INCLUDE(DocStrings.txt)

FUNCTION(ADAPT_DOCSTRING FUNCTIONNAME DOCSTRINGRAW)
  STRING(REPLACE "\n" "\\n" TMPBUFFER "${DOCSTRINGRAW}")
  SET(DOCSTRING_boost_${FUNCTIONNAME} "${TMPBUFFER}" CACHE INTERNAL "" FORCE)
  SET(DOCSTRING_python_${FUNCTIONNAME} "${TMPBUFFER}" CACHE INTERNAL "" FORCE)
ENDFUNCTION()

FOREACH(FUNCTIONAME ${DOCSTRING_ALL_FUNCTION})
  ADAPT_DOCSTRING("${FUNCTIONAME}" "${DOCSTRINGRAW_${FUNCTIONAME}}")
ENDFOREACH()
